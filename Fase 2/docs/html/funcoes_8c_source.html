190bool CalcularInterferencias(Grafo* grafo, FILE* saida) {
191 if (grafo == NULL || saida == NULL) return false;
192
193 fprintf(saida, "=== PONTOS DE INTERFERENCIA ===\n");
194
195 for (int i = 0; i < grafo->total_antenas; i++) {
196 for (int j = i + 1; j < grafo->total_antenas; j++) {
197 if (grafo->antenas[i].frequencia == grafo->antenas[j].frequencia) {
198 int dx = grafo->antenas[j].coluna - grafo->antenas[i].coluna;
199 int dy = grafo->antenas[j].linha - grafo->antenas[i].linha;
200
201 if (dx == 0 || dy == 0 || abs(dx) == abs(dy)) {
202 for (int k = 1; k < 3; k++) {
203 int x = grafo->antenas[i].coluna + (k * dx) / 3;
204 int y = grafo->antenas[i].linha + (k * dy) / 3;
205
206 if (x >= 0 && x < MAX_COLUNAS && y >= 0 && y < MAX_LINHAS) {
207 fprintf(saida, "(%d,%d)\n", x, y);
208 }
209 }
210 }
211 }
212 }
213 }
