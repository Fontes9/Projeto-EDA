62Grafo CarregarAntenasDoFicheiro(const char* nome_ficheiro) {
63 Grafo grafo = { NULL, 0 };
64
65 if (!nome_ficheiro) return grafo;
66
67 FILE* ficheiro = fopen(nome_ficheiro, "r");
68 if (!ficheiro) {
69 perror("Erro ao abrir ficheiro");
70 return grafo;
71 }
72
73 int num_linhas, num_colunas; // Renomeei para evitar conflito com Antena->linha
74 if (fscanf(ficheiro, "%d %d", &num_linhas, &num_colunas) != 2) {
75 fclose(ficheiro);
76 fprintf(stderr, "Formato de ficheiro inválido\n");
77 return grafo;
78 }
79
80 while (fgetc(ficheiro) != '\n'); // Consome o restante da primeira linha
81
82 char buffer_linha[1024]; // Buffer para ler cada linha do ficheiro
83 for (int y = 0; y < num_linhas && fgets(buffer_linha, sizeof(buffer_linha), ficheiro) != NULL; y++) {
84 buffer_linha[strcspn(buffer_linha, "\n")] = '\0'; // Remove newline
85 for (int x = 0; x < num_colunas && x < (int)strlen(buffer_linha); x++) {
86 if (buffer_linha[x] != '.' && buffer_linha[x] != ' ') {
87 AdicionarAntena(&grafo, buffer_linha[x], x, y);
88 }
89 }
90 }
91
92 fclose(ficheiro);
93
94 // Cria as adjacências entre antenas da mesma frequência
95 for (Antena* a1 = grafo.antenas; a1 != NULL; a1 = a1->proxima) {
96 for (Antena* a2 = grafo.antenas; a2 != NULL; a2 = a2->proxima) {
97 if (a1 != a2 && a1->frequencia == a2->frequencia) {
98 AdicionarAdjacencia(a1, a2);
99 }
100 }
101 }
102
103 return grafo;
104}

134bool ProcuraEmLargura(Grafo* grafo, Antena* inicio, FILE* saida) {

141 if (!novo) return false;

144
145 frente = tras = novo;
146 inicio->visitada = true;

155
156 for (Adjacencia* adj = atual->conexoes; adj != NULL; adj = adj->proxima) {
157 if (!adj->destino->visitada) {
158 adj->destino->visitada = true;
159 FilaNode* novo_node = malloc(sizeof(FilaNode));
160 if (!novo_node) continue;
161 novo_node->antena = adj->destino;
162 novo_node->proxima = NULL;
163
164 if (!frente) frente = novo_node;
165 else tras->proxima = novo_node;
166 tras = novo_node;
167 }
168 }
169 }
170 return true;

274bool CalcularInterferencias(Grafo* grafo, FILE* saida) {
275 if (!grafo || !saida) return false;
276
277 fprintf(saida, "=== PONTOS DE INTERFERENCIA ===\n");
278
279 for (Antena* a1 = grafo->antenas; a1 != NULL; a1 = a1->proxima) {
280 for (Antena* a2 = grafo->antenas; a2 != NULL; a2 = a2->proxima) {
281 if (a1->frequencia == a2->frequencia) {
282 int dx = a2->coluna - a1->coluna;
283 int dy = a2->linha - a1->linha;
284
285 if (dx == 0 || dy == 0 || abs(dx) == abs(dy)) {
286 for (int k = 1; k < 3; k++) {
287 int x = a1->coluna + (k * dx) / 3;
288 int y = a1->linha + (k * dy) / 3;
289 fprintf(saida, "(%d,%d)\n", x, y);
290 }
291 }
292 }
293 }
294 }
295 return true;
